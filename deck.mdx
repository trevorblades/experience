```jsx
import React, {useState} from 'react';

export default function CounterButton() {
  const [count, setCount] = useState(0);

  function increment() {
    setCount(prevCount => prevCount + 1);
  }

  return (
    <button onClick={increment}>
      Clicked {count} times
    </button>
  );
}
```
